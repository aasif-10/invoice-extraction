# Function to generate sample data if no results file available
def generate_sample_data(n=100):
    """Generate sample invoice extraction results"""
    states = ['PUNJAB', 'HARYANA', 'UTTAR PRADESH', 'MAHARASHTRA', 'RAJASTHAN', 
              'MADHYA PRADESH', 'GUJARAT', 'KARNATAKA', 'TAMIL NADU']
    models = ['TAFE MF 241', 'Mahindra 575 DI', 'SONALIKA DI 35 RX', 'John Deere 5050D', 
              'New Holland 3630 TX', 'Escorts Powertrac 439']
    
    results = []
    for i in range(n):
        state = np.random.choice(states)
        model = np.random.choice(models)
        
        # Simulate language mix (30% Hindi/Mixed)
        if np.random.random() < 0.3:
            dealer = f"{state} " if np.random.random() < 0.5 else f"{state} TRACTORS PVT LTD"
        else:
            dealer = f"{state} TRACTORS" if np.random.random() < 0.7 else f"{state} MOTORS"
        
        # Simulate some missing fields and errors
        confidence = np.random.uniform(0.65, 0.99)
        has_dealer = np.random.random() > 0.05  # 95% success
        has_model = np.random.random() > 0.08   # 92% success
        has_hp = np.random.random() > 0.10      # 90% success
        has_cost = np.random.random() > 0.12    # 88% success
        
        results.append({
            'doc_id': f'invoice_{i:03d}',
            'fields': {
                'dealer_name': dealer if has_dealer else None,
                'model_name': model if has_model else None,
                'horse_power': int(np.random.uniform(35, 75)) if has_hp else None,
                'asset_cost': int(np.random.uniform(400000, 1200000)) if has_cost else None,
                'signature': {'present': np.random.random() > 0.02, 'bbox': [100, 200, 150, 80]},
                'stamp': {'present': True, 'bbox': [300, 400, 120, 100]}
            },
            'confidence': confidence,
            'processing_time_sec': np.random.uniform(22, 29.5),
            'cost_estimate_usd': 0.002
        })
    
    return results

# Extract state from dealer name
def extract_state(dealer_name):
    """Extract state from dealer name"""
    if pd.isna(dealer_name):
        return 'Unknown'
    
    state_keywords = {
        'PUNJAB': ['PUNJAB', 'AMRITSAR', 'LUDHIANA'],
        'HARYANA': ['HARYANA', 'KARNAL', 'PANIPAT'],
        'UTTAR PRADESH': ['UTTAR PRADESH', 'UP', 'LUCKNOW'],
        'MAHARASHTRA': ['MAHARASHTRA', 'MUMBAI', 'PUNE'],
        'RAJASTHAN': ['RAJASTHAN', 'JAIPUR', 'JODHPUR'],
        'MADHYA PRADESH': ['MADHYA PRADESH', 'MP', 'BHOPAL'],
        'GUJARAT': ['GUJARAT', 'AHMEDABAD'],
        'TAMIL NADU': ['TAMIL', 'CHENNAI'],
        'KARNATAKA': ['KARNATAKA', 'BANGALORE']
    }
    
    dealer_upper = str(dealer_name).upper()
    for state, keywords in state_keywords.items():
        if any(kw in dealer_upper for kw in keywords):
            return state
    return 'Other'

# Detect language
def detect_language(text):
    """Detect if text contains Hindi/Devanagari characters"""
    if pd.isna(text):
        return 'Unknown'
    hindi_pattern = re.compile(r'[\u0900-\u097F]')
    if hindi_pattern.search(str(text)):
        return 'Hindi/Mixed'
    return 'English'

# Apply feature engineering
df['state'] = df['dealer_name'].apply(extract_state)
df['language'] = df['dealer_name'].apply(detect_language)

# Field completeness
df['fields_extracted'] = (
    df['dealer_name'].notna().astype(int) +
    df['model_name'].notna().astype(int) +
    df['horse_power'].notna().astype(int) +
    df['asset_cost'].notna().astype(int)
)

# Error indicators
df['has_error'] = (df['confidence'] < 0.8) | (df['fields_extracted'] < 4)
df['confidence_category'] = pd.cut(df['confidence'], 
                                    bins=[0, 0.7, 0.8, 0.9, 1.0],
                                    labels=['Low (<70%)', 'Medium (70-80%)', 'Good (80-90%)', 'Excellent (90%+)'])
